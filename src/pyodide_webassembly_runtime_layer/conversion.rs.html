11/// Converts a Rust type to Python
12pub trait ToPy {
13    /// Convert this value to Python
14    fn to_py(&self, py: Python) -> Py<PyAny>;
15}
16
17impl ToPy for Value<Engine> {
18    fn to_py(&self, py: Python) -> Py<PyAny> {
19        fn into_pyobject_infallible<
20            'py,
21            T: IntoPyObject<'py, Output = Bound<'py, S>, Error = Infallible>,
22            S,
23        >(
24            py: Python<'py>,
25            x: T,
26        ) -> Py<PyAny> {
27            match x.into_pyobject(py) {
28                Ok(x) => x.into_any().unbind(),
29                Err(e) => match e {},
30            }
31        }
32
33        #[cfg(feature = "tracing")]
34        tracing::trace!(ty = ?self.ty(), "Value::to_py");
35
36        match self {
37            Self::I32(v) => into_pyobject_infallible(py, v),
38            // WebAssembly explicitly requires all i64's to be a BigInt
39            // Pyodide auto-converts BigInts, so we wrap it in an Object
40            Self::I64(v) => i64_to_js_bigint(py, *v).unbind(),
41            Self::F32(v) => into_pyobject_infallible(py, v),
42            Self::F64(v) => into_pyobject_infallible(py, v),
43            Self::FuncRef(None) | Self::ExternRef(None) => py.None(),
44            Self::FuncRef(Some(func)) => func.to_py(py),
45            Self::ExternRef(Some(r#ref)) => r#ref.to_py(py),
46        }
47    }
48}

50impl ToPy for Extern<Engine> {
51    fn to_py(&self, py: Python) -> Py<PyAny> {
52        #[cfg(feature = "tracing")]
53        tracing::trace!("Extern::to_py");
54
55        match self {
56            Self::Global(v) => v.to_py(py),
57            Self::Table(v) => v.to_py(py),
58            Self::Memory(v) => v.to_py(py),
59            Self::Func(v) => v.to_py(py),
60        }
61    }
62}

64pub trait ValueExt: Sized {
65    /// Convert a value to its type
66    fn ty(&self) -> ValueType;
67
68    /// Convert the [`PyAny`] value into a Value of the supplied type
69    fn from_py_typed(value: Bound<PyAny>, ty: ValueType) -> Result<Self, PyErr>;
70}
71
72impl ValueExt for Value<Engine> {
73    /// Convert a value to its type
74    fn ty(&self) -> ValueType {
75        match self {
76            Self::I32(_) => ValueType::I32,
77            Self::I64(_) => ValueType::I64,
78            Self::F32(_) => ValueType::F32,
79            Self::F64(_) => ValueType::F64,
80            Self::FuncRef(_) => ValueType::FuncRef,
81            Self::ExternRef(_) => ValueType::ExternRef,
82        }
83    }
84
85    fn from_py_typed(value: Bound<PyAny>, ty: ValueType) -> Result<Self, PyErr> {
86        match ty {
87            ValueType::I32 => Ok(Self::I32(value.extract()?)),
88            // Try to unwrap a number, BigInt, or Object-wrapped BigInt
89            ValueType::I64 => Ok(Self::I64(try_i64_from_js_bigint(value)?)),
90            ValueType::F32 => Ok(Self::F32(value.extract()?)),
91            ValueType::F64 => Ok(Self::F64(value.extract()?)),
92            ValueType::ExternRef => {
93                if value.is_none() {
94                    Ok(Self::ExternRef(None))
95                } else {
96                    Ok(Self::ExternRef(Some(ExternRef::from_exported_externref(
97                        value,
98                    ))))
99                }
100            },
101            ValueType::FuncRef => {
102                if value.is_none() {
103                    Ok(Self::FuncRef(None))
104                } else {
105                    Err(PyRuntimeError::new_err(
106                        "conversion to a function outside of a module export is not permitted as \
107                         its type signature is unknown",
108                    ))
109                }
110            },
111        }
112    }
113}

115pub trait ValueTypeExt {
116    /// Converts this type into the canonical ABI kind
117    ///
118    /// See: <https://webassembly.github.io/spec/js-api/#globals>
119    fn as_js_descriptor(&self) -> &str;
120}
121
122impl ValueTypeExt for ValueType {
123    fn as_js_descriptor(&self) -> &str {
124        match self {
125            Self::I32 => "i32",
126            Self::I64 => "i64",
127            Self::F32 => "f32",
128            Self::F64 => "f64",
129            Self::FuncRef => "anyfunc",
130            Self::ExternRef => "externref",
131        }
132    }
133}

135fn i64_to_js_bigint(py: Python, v: i64) -> Bound<PyAny> {
136    fn object_wrapped_bigint(py: Python) -> Result<&Bound<PyAny>, PyErr> {
137        static OBJECT_WRAPPED_BIGINT: GILOnceCell<Py<PyAny>> = GILOnceCell::new();
138
139        OBJECT_WRAPPED_BIGINT
140            .get_or_try_init(py, || {
141                Ok(py
142                    .import(intern!(py, "pyodide"))?
143                    .getattr(intern!(py, "code"))?
144                    .getattr(intern!(py, "run_js"))?
145                    .call1((
146                        "function objectWrappedBigInt(v){ return Object(BigInt(v)); } \
147                         objectWrappedBigInt",
148                    ))?
149                    .unbind())
150            })
151            .map(|x| x.bind(py))
152    }
153
154    let bigint = (|| object_wrapped_bigint(py)?.call1((v,)))();
155
156    bigint.expect("conversion from i64 to Object(BigInt(v)) should not fail")
157}

159fn try_i64_from_js_bigint(v: Bound<PyAny>) -> Result<i64, PyErr> {
160    fn js_bigint(py: Python) -> Result<&Bound<PyAny>, PyErr> {
161        static JS_BIG_INT: GILOnceCell<Py<PyAny>> = GILOnceCell::new();
162        JS_BIG_INT.import(py, "js", "BigInt")
163    }
164
165    // First wrap inside a BigInt to force coersion, then try to convert into an i64
166    js_bigint(v.py())?.call1((v,))?.extract()
167}

174/// Check if `object` is an instance of the JavaScript class with `constructor`.
175pub fn instanceof(object: &Bound<PyAny>, constructor: &Bound<PyAny>) -> Result<bool, PyErr> {
176    fn is_instance_of(py: Python) -> Result<&Bound<PyAny>, PyErr> {
177        static IS_INSTANCE_OF: GILOnceCell<Py<PyAny>> = GILOnceCell::new();
178
179        IS_INSTANCE_OF
180            .get_or_try_init(py, || {
181                Ok(py
182                    .import(intern!(py, "pyodide"))?
183                    .getattr(intern!(py, "code"))?
184                    .getattr(intern!(py, "run_js"))?
185                    .call1((
186                        "function isInstanceOf(object, constructor){ return (object instanceof \
187                         constructor); } isInstanceOf",
188                    ))?
189                    .unbind())
190            })
191            .map(|x| x.bind(py))
192    }
193
194    is_instance_of(object.py())?
195        .call1((object, constructor))?
196        .extract()
197}

199pub fn create_js_object(py: Python) -> Result<Bound<PyAny>, PyErr> {
200    fn js_object_new(py: Python) -> Result<&Bound<PyAny>, PyErr> {
201        static JS_OBJECT_NEW: GILOnceCell<Py<PyAny>> = GILOnceCell::new();
202        JS_OBJECT_NEW.import(py, "js.Object", "new")
203    }
204
205    js_object_new(py)?.call0()
206}

208pub fn py_to_js_proxy<T>(object: Bound<T>) -> Result<Bound<PyAny>, PyErr> {
209    fn to_js(py: Python) -> Result<&Bound<PyAny>, PyErr> {
210        static TO_JS: GILOnceCell<Py<PyAny>> = GILOnceCell::new();
211        TO_JS.import(py, "pyodide.ffi", "to_js")
212    }
213
214    let py = object.py();
215
216    to_js(py)?.call(
217        (object,),
218        Some(&[(intern!(py, "create_pyproxies"), true)].into_py_dict(py)?),
219    )
