8#[derive(Debug, Clone, Copy, PartialEq, Eq)]
9pub struct UnsupportedWasmFeatureExtensionError {
10    pub required: FlagSet<WasmFeatureExtension>,
11    pub supported: FlagSet<WasmFeatureExtension>,
12}
13
14impl UnsupportedWasmFeatureExtensionError {
15    pub fn check_support(py: Python, bytes: &[u8]) -> Result<Result<(), Self>, PyErr> {
16        let err = Self {
17            required: WasmFeatureExtension::required(bytes),
18            supported: *WasmFeatureExtension::supported(py)?,
19        };
20
21        if (err.required & (!err.supported)).is_empty() {
22            return Ok(Ok(()));
23        }
24
25        Ok(Err(err))
26    }
27}
28
29impl fmt::Display for UnsupportedWasmFeatureExtensionError {
30    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
31        writeln!(
32            fmt,
33            "A WASM module requires the following feature extensions, which are not supported by \
34             your browser:"
35        )?;
36        writeln!(fmt)?;
37
38        for missing in self.required & (!self.supported) {
39            writeln!(fmt, " - {missing}")?;
40        }
41
42        writeln!(fmt)?;
43        writeln!(
44            fmt,
45            "Please check out https://webassembly.org/features/ to see if a newer version of your \
46             browser already supports these features."
47        )
48    }
49}

98impl WasmFeatureExtension {
99    #[allow(clippy::too_many_lines)]
100    #[must_use]
101    pub fn required(bytes: &[u8]) -> FlagSet<Self> {
102        let mut required: FlagSet<_> = FlagSet::default();
103
104        required.extend(
105            FlagSet::<Self>::full()
106                .into_iter()
107                .filter(|extension| Self::requires_features(bytes, (*extension).into())),
108        );
109
110        required
111    }
112
113    pub fn supported(py: Python) -> Result<&'static FlagSet<Self>, PyErr> {
114        static SUPPORTED_FEATURES: PyOnceLock<FlagSet<WasmFeatureExtension>> = PyOnceLock::new();
115
116        SUPPORTED_FEATURES.get_or_try_init(py, || {
117            let mut supported = FlagSet::default();
118
119            for extension in FlagSet::<Self>::full() {
120                if extension.check_if_supported(py)? {
121                    supported |= extension;
122                }
123            }
124
125            Ok(supported)
126        })
127    }
128
129    pub fn check_if_supported(self, py: Python) -> Result<bool, PyErr> {
130        let canary = self.canary_bytes();
131
132        if matches!(self, Self::MultiMemory) {
133            Self::try_create_wasm_module_from_bytes(py, canary)
134        } else {
135            Self::try_validate_wasm_bytes(py, canary)
136        }
137    }
138
139    const fn canary_bytes(self) -> &'static [u8] {
140        // The WASM feature detection mechanism and the detector WASM modules
141        // are adapted from the Google Chrome Team's `wasm-feature-detect`
142        // repository, which is released under the Apache-2.0 License.
143        // https://github.com/GoogleChromeLabs/wasm-feature-detect/tree/8bfe6691b0749b53d605f3220f15e68751c4b5b6/src/detectors
144        //
145        // The detector modules have been compiled from *.wat to *.wasm
146        // using binaryen's wasm-as.
147        match self {
148            Self::BulkMemory => include_bytes!("bulk-memory.wasm"),
149            Self::Exceptions => include_bytes!("exceptions.wasm"),
150            Self::ExtendedConst => include_bytes!("extended-const.wasm"),
151            Self::FunctionReferences => include_bytes!("function-references.wasm"),
152            Self::GC => include_bytes!("gc.wasm"),
153            Self::Memory64 => include_bytes!("memory64.wasm"),
154            Self::MultiMemory => include_bytes!("multi-memory.wasm"),
155            Self::MultiValue => include_bytes!("multi-value.wasm"),
156            Self::MutableGlobal => include_bytes!("mutable-global.wasm"),
157            Self::ReferenceTypes => include_bytes!("reference-types.wasm"),
158            Self::RelaxedSimd => include_bytes!("relaxed-simd.wasm"),
159            Self::SaturatingFloatToInt => include_bytes!("saturating-float-to-int.wasm"),
160            Self::SignExtension => include_bytes!("sign-extension.wasm"),
161            Self::Simd => include_bytes!("simd.wasm"),
162            Self::TailCall => include_bytes!("tail-call.wasm"),
163            Self::Threads => include_bytes!("threads.wasm"),
164        }
165    }
166
167    fn requires_features(bytes: &[u8], features: wasmparser::WasmFeatures) -> bool {
168        wasmparser::Validator::new_with_features(!features)
169            .validate_all(bytes)
170            .is_err()
171    }
172
173    fn try_validate_wasm_bytes(py: Python, bytes: &[u8]) -> Result<bool, PyErr> {
174        let buffer = js_uint8_array_new(py)?.call1((bytes,))?;
175        let valid = web_assembly_validate(py)?.call1((buffer,))?.extract()?;
176        Ok(valid)
177    }
178
179    fn try_create_wasm_module_from_bytes(py: Python, bytes: &[u8]) -> Result<bool, PyErr> {
180        let buffer = js_uint8_array_new(py)?.call1((bytes,))?;
181        let module = web_assembly_module_new(py)?.call1((buffer,));
182        Ok(module.is_ok())
183    }
184}

223#[cfg(test)]
224mod tests {
225    use super::*;
226
227    #[test]
228    fn required_features() {
229        for feature in FlagSet::<WasmFeatureExtension>::full() {
230            let required = WasmFeatureExtension::required(feature.canary_bytes());
231
232            let check = match feature {
233                // the function-references feature depends on reference-types
234                // FIXME: remove the dependency on bulk-memory
235                WasmFeatureExtension::FunctionReferences => {
236                    FlagSet::from(feature)
237                        | WasmFeatureExtension::ReferenceTypes
238                        | WasmFeatureExtension::BulkMemory
239                },
240                // the relaxed-simd feature depends on simd
241                WasmFeatureExtension::RelaxedSimd => {
242                    FlagSet::from(feature) | WasmFeatureExtension::Simd
243                },
244                // otherwise, every feature should only require itself
245                _ => FlagSet::from(feature),
246            };
247
248            assert_eq!(
249                required, check,
250                "{feature} should only require {check:?}, but needs {required:?}"
251            );
252        }
253    }
